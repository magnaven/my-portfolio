    Step 1 — Create `providers/SmoothScrollProvider.tsx`:

    This component MUST:
    - Be a "use client" component (Lenis and GSAP use browser APIs)
    - Import ReactLenis from `lenis/react` (NOT `@studio-freight/react-lenis` — deprecated)
    - Register ScrollTrigger and useGSAP plugins in one place
    - Drive Lenis from `gsap.ticker.add()` with `autoRaf: false` — one RAF loop, no conflicts with ScrollTrigger

    ```typescript
    "use client";

    import { ReactLenis } from "lenis/react";
    import { useEffect, useRef } from "react";
    import gsap from "gsap";
    import ScrollTrigger from "gsap/ScrollTrigger";
    import { useGSAP } from "@gsap/react";

    // Register all GSAP plugins in one place
    gsap.registerPlugin(ScrollTrigger, useGSAP);

    interface Props {
      children: React.ReactNode;
    }

    export function SmoothScrollProvider({ children }: Props) {
      const lenisRef = useRef<{ lenis?: { raf: (time: number) => void } } | null>(null);

      useEffect(() => {
        function update(time: number) {
          lenisRef.current?.lenis?.raf(time * 1000);
        }

        gsap.ticker.add(update);
        gsap.ticker.lagSmoothing(0);

        return () => {
          gsap.ticker.remove(update);
        };
      }, []);

      return (
        <ReactLenis root options={{ autoRaf: false }} ref={lenisRef}>
          {children}
        </ReactLenis>
      );
    }
    ```

    Step 2 — Create `app/template.tsx`:

    This MUST be `template.tsx` (not `layout.tsx`). Next.js remounts template on every navigation — that is the lifecycle that Motion's animate/exit props depend on. Layout persists and will not trigger exit animations.

    ```typescript
    "use client";

    import { motion } from "motion/react";

    export default function Template({ children }: { children: React.ReactNode }) {
      return (
        <motion.div
          initial={{ opacity: 0, y: 8 }}
          animate={{ opacity: 1, y: 0 }}
          exit={{ opacity: 0, y: -8 }}
          transition={{ duration: 0.35, ease: "easeInOut" }}
        >
          {children}
        </motion.div>
      );
    }
    ```

    Both files must have `"use client"` as the very first line — they use browser APIs that are unavailable in Server Components.
